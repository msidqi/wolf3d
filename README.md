installation: 
Place your SDL2 libraries in
	src/libSDL/libSDL2-2.0.0.dylib
	src/libSDL_image/libSDL2_image-2.0.0.dylib
	src/libSDL_ttf/libSDL2_ttf-2.0.0.dylib
	src/libSDL_mixer/libSDL2_mixer-2.0.0.dylib


make && ./wolf3d
